213  inline void tokenize(const std::string &inputString, StringVector &tokens, const std::string &delimiter = " ")
214  {
215  std::string::size_type lastPos = inputString.find_first_not_of(delimiter, 0);
216  std::string::size_type pos = inputString.find_first_of(delimiter, lastPos);
217 
218  while ((std::string::npos != pos) || (std::string::npos != lastPos))
219  {
220  tokens.push_back(inputString.substr(lastPos, pos - lastPos));
221  lastPos = inputString.find_first_not_of(delimiter, pos);
222  pos = inputString.find_first_of(delimiter, lastPos);
223  }
224  }
