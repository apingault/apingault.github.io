198  StatusCode Archiver::recursiveFill(Directory<MonitorElement> *pDirectory, TDirectory *pROOTDir)
199  {
200  if(nullptr == pDirectory || nullptr == pROOTDir)
201  return STATUS_CODE_INVALID_PTR;
202 
203  pROOTDir->cd();
204 
205  const std::vector<Directory<MonitorElement>*> &subDirList(pDirectory->subdirs());
206 
207  if(!subDirList.empty())
208  {
209  for(auto iter = subDirList.begin(), endIter = subDirList.end() ; endIter != iter ; ++iter)
210  {
211  auto *pSubDir = *iter;
212  TDirectory *pROOTSubDir = pROOTDir->mkdir(pSubDir->name().c_str());
213 
214  if(nullptr != pROOTSubDir)

216  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, Archiver::recursiveFill(pSubDir, pROOTSubDir));
