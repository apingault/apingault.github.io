137  StatusCode XmlHelper::performConstantReplacement(std::string &value, const StringMap &constants)
138  {
139  size_t pos = value.find("${") ;
140  std::string original(value);
141 
142  while( pos != std::string::npos ) {
143 
144  size_t pos2 = value.find_first_of( "}", pos+2 ) ;
145 
146  if( pos2 == std::string::npos ) {
147  dqm_error( "XMLHelper::performConstantReplacement : couldn't parse constant value !" ) ;
148  return STATUS_CODE_FAILURE;
149  }
150 
151  const std::string key( value.substr( pos+2 , pos2-pos-2 )) ;
152  auto findConstant = constants.find( key ) ;
153  const std::string replacementValue( findConstant != constants.end() ? findConstant->second : "" ) ;
154 
155  if( replacementValue.empty() ) {
156  dqm_error( "XMLParser::performConstantReplacement : constant \"{0}\" not found in available constants !", key );
157  return STATUS_CODE_FAILURE;
158  }
159 
160  value.replace( pos , (pos2+1-pos) , replacementValue ) ;
161  pos2 = pos + replacementValue.size() ; // new end position after replace
162 
163  pos = value.find("${", pos2) ; // find next possible key to replace
164  }
165 
166  return STATUS_CODE_SUCCESS;
167  }

172  {

174 
175  while( 1 )

184  }
185 
186  if( child->Value() != std::string("include") ) {

195  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::processIncludeElement( fileName, child , constants , document ) );
196

209  child = includeAfter->NextSiblingElement() ;
210  }
211

385  StatusCode XmlHelper::replaceAllXmlAttributes(TiXmlElement *pXmlElement, const ParameterMap &parameters)
386  {
387  for(auto paramIter = parameters.begin(), endParamIter = parameters.end() ;
388  endParamIter != paramIter ; ++paramIter)
389  {
390  std::string key(paramIter->first);
391  std::string value(paramIter->second);
392 
393  StringVector xmlElementNames;
394  dqm4hep::core::tokenize(key, xmlElementNames, ".");
395 
396  TiXmlElement *pCurrentElement = pXmlElement;
397  bool keyIsParameter = false;
398 
399  for(unsigned int e=0 ; e<xmlElementNames.size() ; e++)
400  {
401  std::string subKey(xmlElementNames.at(e));
402  unsigned int eltId = 0;
403 
404  size_t posStart = subKey.find("[");
405 
406  if( posStart != std::string::npos )
407  {
408  if( subKey.back() != ']' )
409  {
410  dqm_error( "Missing ']' char in '{0}'", subKey );
411  return STATUS_CODE_NOT_FOUND;
412  }
413 
414  std::string eltIdStr = subKey.substr( posStart+1 , subKey.size() - (posStart+1) -1 );
415 
416  if( ! dqm4hep::core::stringToType( eltIdStr , eltId ) )
417  {
418  dqm_error( "Couldn't convert '{0}' to array id in key '{1}' !", eltIdStr, subKey);
419  return STATUS_CODE_FAILURE;
420  }
421 
422  subKey = subKey.substr( 0 , posStart );
423  }
424 
425  // last sub key
426  if( e == xmlElementNames.size()-1 )
427  {
428  if( keyIsParameter )
429  {
430  if( ! pCurrentElement->NoChildren() )
431  pCurrentElement->Clear();
432 
433  TiXmlText *pText = new TiXmlText( value );
434  pCurrentElement->LinkEndChild(pText);
435 
436  dqm_debug( "Parameter key '{0}' set to {1}", key, value );
437  }
438  else
439  {
440  pCurrentElement->SetAttribute( subKey , value );
441  dqm_debug( "Key attribute '{0}' set to ", key, value );
442  }
443  }
444  else
445  {
446  if( e == xmlElementNames.size()-2 && subKey == "parameter" )
447  {
448  bool found = false;
449  std::string parameterName( xmlElementNames.at(e+1) );
450 
451  for (TiXmlElement *pXmlElement = pCurrentElement->FirstChildElement("parameter"); NULL != pXmlElement;
452  pXmlElement = pXmlElement->NextSiblingElement("parameter"))
453  {
454  std::string name;
455  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(pXmlElement, "name", name));
456 
457  if( name == parameterName )
458  {
459  pCurrentElement = pXmlElement;
460  found = true;
461  break;
462  }
463  }
464 
465  if( ! found )
466  {
467  dqm_error( "Key '{0}', subKey '{1}' xml element not found !", key, subKey );
468  return STATUS_CODE_NOT_FOUND;
469  }
470 
471  keyIsParameter = true;
472  }
473  else
474  {
475  unsigned int currentElt = 0;
476  TiXmlElement *pElt = pCurrentElement->FirstChildElement( subKey );
477 
478  while( pElt && currentElt < eltId )
479  {
480  pElt = pElt->NextSiblingElement( subKey );
481  currentElt++;
482  }
483 
484  pCurrentElement = pElt;
485  }
486 
487  if( NULL == pCurrentElement )
488  {
489  dqm_error( "Key '{0}', subKey '{1}' xml element not found !", key, subKey );
490  return STATUS_CODE_NOT_FOUND;
491  }
492  }
493  }
494  }
495 
496  return STATUS_CODE_SUCCESS;
497  }
