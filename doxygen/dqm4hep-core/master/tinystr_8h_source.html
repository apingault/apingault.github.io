64  TiXmlString is an emulation of a subset of the std::string template.
65  Its purpose is to allow compiling TinyXML on compilers with no or poor STL support.
66  Only the member functions relevant to the TinyXML project have been implemented.
67  The buffer allocation is made by a simplistic power of 2 like mechanism : if we increase
68  a string and there's no more room, we allocate a buffer twice as big as we need.

234  void init(size_type sz, size_type cap)
235  {
236  if (cap)
237  {
238  // Lee: the original form:
239  // rep_ = static_cast<Rep*>(operator new(sizeof(Rep) + cap));
240  // doesn't work in some cases of new being overloaded. Switching
241  // to the normal allocation, although use an 'int' for systems
242  // that are overly picky about structure alignment.
243  const size_type bytesNeeded = sizeof(Rep) + cap;
244  const size_type intsNeeded = ( bytesNeeded + sizeof(int) - 1 ) / sizeof( int );
245  rep_ = reinterpret_cast<Rep*>( new int[ intsNeeded ] );
246 
247  rep_->str[ rep_->size = sz ] = '\0';
248  rep_->capacity = cap;
249  }
250  else
251  {
252  rep_ = &nullrep_;
253  }
254  }
