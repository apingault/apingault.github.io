56  void TiXmlBase::EncodeString( const TIXML_STRING& str, TIXML_STRING* outString )
57  {
58  int i=0;
59 
60  while( i<(int)str.length() )
61  {
62  unsigned char c = (unsigned char) str[i];
63 
64  if ( c == '&'
65  && i < ( (int)str.length() - 2 )
66  && str[i+1] == '#'
67  && str[i+2] == 'x' )
68  {
69  // Hexadecimal character reference.
70  // Pass through unchanged.
71  // &#xA9; -- copyright symbol, for example.
72  //
73  // The -1 is a bug fix from Rob Laveaux. It keeps
74  // an overflow from happening if there is no ';'.
75  // There are actually 2 ways to exit this loop -
76  // while fails (error case) and break (semicolon found).
77  // However, there is no mechanism (currently) for
78  // this function to return an error.
79  while ( i<(int)str.length()-1 )
80  {
81  outString->append( str.c_str() + i, 1 );
82  ++i;
83  if ( str[i] == ';' )
84  break;
85  }
86  }
87  else if ( c == '&' )
88  {
89  outString->append( entity[0].str, entity[0].strLength );
90  ++i;
91  }
92  else if ( c == '<' )
93  {
94  outString->append( entity[1].str, entity[1].strLength );
95  ++i;
96  }
97  else if ( c == '>' )
98  {
99  outString->append( entity[2].str, entity[2].strLength );
100  ++i;
101  }
102  else if ( c == '\"' )
103  {
104  outString->append( entity[3].str, entity[3].strLength );
105  ++i;
106  }
107  else if ( c == '\'' )
108  {
109  outString->append( entity[4].str, entity[4].strLength );
110  ++i;
111  }
112  else if ( c < 32 )
113  {
114  // Easy pass at non-alpha/numeric/symbol
115  // Below 32 is symbolic.
116  char buf[ 32 ];
117 
118 #if defined(TIXML_SNPRINTF)
119  TIXML_SNPRINTF( buf, sizeof(buf), "&#x%02X;", (unsigned) ( c & 0xff ) );
120 #else
121  sprintf( buf, "&#x%02X;", (unsigned) ( c & 0xff ) );
122 #endif
123 
124  //*ME: warning C4267: convert 'size_t' to 'int'
125  //*ME: Int-Cast to make compiler happy ...
126  outString->append( buf, (int)strlen( buf ) );
127  ++i;
128  }
129  else
130  {
131  //char realc = (char) c;
132  //outString->append( &realc, 1 );
133  *outString += (char) c; // somewhat more efficient function call.
134  ++i;
135  }
136  }

766  }

955  location.Clear();

958  long length = 0;
959  fseek( file, 0, SEEK_END );
960  length = ftell( file );
961  fseek( file, 0, SEEK_SET );

970  // Subtle bug here. TinyXml did use fgets. But from the XML spec:
971  // 2.11 End-of-Line Handling
972  // <snip>
973  // <quote>
974  // ...the XML processor MUST behave as if it normalized all line breaks in external
975  // parsed entities (including the document entity) on input, before parsing, by translating
976  // both the two-character sequence #xD #xA and any #xD that is not followed by #xA to
977  // a single #xA character.
978  // </quote>
979  //
980  // It is not clear fgets does that, and certainly isn't clear it works cross platform.
981  // Generally, you expect fgets to translate from the convention of the OS to the c/unix
982  // convention, and not work generally.
983 
984  /*
985  while( fgets( buf, sizeof(buf), file ) )
986  {
987  data += buf;
988  }
989  */
990 
991  char* buf = new char[ length+1 ];
992  buf[0] = 0;
993 
994  if ( fread( buf, length, 1, file ) != 1 ) {
995  delete [] buf;
996  SetError( TIXML_ERROR_OPENING_FILE, 0, 0, TIXML_ENCODING_UNKNOWN );
997  return false;
998  }

1000  // Process the buffer in place to normalize new lines. (See comment above.)
1001  // Copies from the 'p' to 'q' pointer, where p can advance faster if
1002  // a newline-carriage return is hit.
1003  //
1004  // Wikipedia:
1005  // Systems based on ASCII or a compatible character set use either LF (Line feed, '\n', 0x0A, 10 in decimal) or
1006  // CR (Carriage return, '\r', 0x0D, 13 in decimal) individually, or CR followed by LF (CR+LF, 0x0D 0x0A)...
1007  // * LF: Multics, Unix and Unix-like systems (GNU/Linux, AIX, Xenix, Mac OS X, FreeBSD, etc.), BeOS, Amiga, RISC OS, and others
1008  // * CR+LF: DEC RT-11 and most other early non-Unix, non-IBM OSes, CP/M, MP/M, DOS, OS/2, Microsoft Windows, Symbian OS
1009  // * CR: Commodore 8-bit machines, Apple II family, Mac OS up to version 9 and OS-9
1010 
1011  const char* p = buf; // the read head
1012  char* q = buf; // the write head
1013  const char CR = 0x0d;
1014  const char LF = 0x0a;
1015 
1016  buf[length] = 0;
1017  while( *p ) {
1018  assert( p < (buf+length) );
1019  assert( q <= (buf+length) );
1020  assert( q <= p );
1021 
1022  if ( *p == CR ) {
1023  *q++ = LF;
1024  p++;
1025  if ( *p == LF ) { // check for CR+LF (and skip LF)
1026  p++;
1027  }
1028  }
1029  else {
1030  *q++ = *p++;
1031  }
1032  }
1033  assert( q <= (buf+length) );
1034  *q = 0;
1035 
1036  Parse( buf, 0, encoding );
1037 
1038  delete [] buf;
1039  return !Error();
1040  }
