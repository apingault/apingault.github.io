274  StatusCode MonitorElementManager::bookObject(const std::string &path, const std::string &name, MonitorElement *&pMonitorElement,
275  allocator_helper<TObject, ObjectType, Args...> allocator, Args ...args)
276  {
277  pMonitorElement = nullptr;
278 
279  if(!ObjectType::Class()->HasDictionary())
280  {
281  dqm_error( "Couldn't book object of type '{0}', because this class has no dictionnary!", ObjectType::Class_name() );
282  return STATUS_CODE_NOT_ALLOWED;
283  }
284 
285  if(ObjectType::Class()->IsForeign())
286  {
287  dqm_error( "Couldn't book object of type '{0}', because this class is foreign (does not have a Streamer method))!", ObjectType::Class_name() );
288  return STATUS_CODE_NOT_ALLOWED;
289  }
290 
291  if(!ObjectType::Class()->InheritsFrom("TNamed"))
292  {
293  dqm_error( "Couldn't book object of type '{0}', because this class doesn't inherit TNamed (required to call SetName())!", ObjectType::Class_name() );
294  return STATUS_CODE_NOT_ALLOWED;
295  }
296 
297  const bool objectStat(TObject::GetObjectStat());
298  TObject::SetObjectStat(false);
299  TObject *pTObject = allocator.create(args...);
300  TObject::SetObjectStat(objectStat);
301 
302  if(!pTObject)
303  {
304  dqm_warning( "Couldn't allocate monitor element of type '{0}', path '{1}', name '{2}'", ObjectType::Class_name(), path, name );
305  return STATUS_CODE_FAILURE;
306  }
307 
308  ((TNamed*)pTObject)->SetName(name.c_str());
309 
310  return this->addMonitorElement(path, pTObject, pMonitorElement);
311  }
