98  StatusCode PluginManager::loadLibrary( const std::string &libraryName )
99  {
100  size_t idx = libraryName.find_last_of( "/" );
101 
102  // the library basename, i.e. /path/to/libBlah.so --> libBlah.so
103  std::string libBaseName( libraryName.substr( idx + 1 ) );
104 
105  char *real_path = realpath( libraryName.c_str(), NULL );
106 
107  if( NULL != real_path )
108  {
109  dqm_info( "<!-- Loading shared library : {0} ({1})-->", real_path, libBaseName );
110 
111  // use real_path
112  free( real_path );
113  }
114  else
115  {
116  dqm_info( "<!-- Loading shared library : {0} ({1})-->", libraryName, libBaseName );
117  }
118 
119  void *pLibPointer = dlopen( libraryName.c_str(), RTLD_LAZY | RTLD_GLOBAL );
120 
121  if( pLibPointer == 0 )
122  {
123  dqm_error( "<!-- ERROR loading shared library : {0}\n -> {1} -->", libraryName, dlerror() );
124  return STATUS_CODE_FAILURE;
125  }
126  else
127  return STATUS_CODE_SUCCESS;

164  StatusCode PluginManager::registerPlugin( Plugin *pPlugin )
165  {
166  // null ptr case
167  if( NULL == pPlugin )
168  return STATUS_CODE_INVALID_PTR;
169 
170  // check if the plug is already registered
171  if( isPluginRegistered( pPlugin->pluginName() ) )
172  {
173  delete pPlugin;
174  pPlugin = NULL;
175  return STATUS_CODE_ALREADY_PRESENT;
176  }
177 
178  // try to register it
179  if( ! m_pluginMap.insert( PluginMap::value_type( pPlugin->pluginName(), pPlugin ) ).second )
180  {
181  delete pPlugin;
182  pPlugin = NULL;
183 
184  return STATUS_CODE_FAILURE;
185  }
186 
187  return STATUS_CODE_SUCCESS;
188  }
